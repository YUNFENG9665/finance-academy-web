import torch
import torchvision.models as models
from torchvision import transforms
from PIL import Image
import numpy as np

class SatelliteImageAnalyzer:
    """卫星图像分析器"""

    def __init__(self):
        # 加载预训练的目标检测模型
        self.detection_model = models.detection.fasterrcnn_resnet50_fpn(
            pretrained=True
        )
        self.detection_model.eval()

        # 图像预处理
        self.transform = transforms.Compose([
            transforms.ToTensor(),
            transforms.Normalize(
                mean=[0.485, 0.456, 0.406],
                std=[0.229, 0.224, 0.225]
            )
        ])

        # COCO数据集类别
        self.coco_classes = {
            3: 'car',
            6: 'bus',
            8: 'truck'
        }

    def count_vehicles(self, image_path):
        """统计停车场车辆数量"""
        image = Image.open(image_path).convert('RGB')
        image_tensor = transforms.ToTensor()(image).unsqueeze(0)

        with torch.no_grad():
            predictions = self.detection_model(image_tensor)

        # 筛选车辆
        boxes = predictions[0]['boxes']
        labels = predictions[0]['labels']
        scores = predictions[0]['scores']

        vehicle_count = 0
        vehicle_details = {'car': 0, 'bus': 0, 'truck': 0}

        for i, (label, score) in enumerate(zip(labels, scores)):
            if score > 0.5 and label.item() in self.coco_classes:
                vehicle_count += 1
                vehicle_type = self.coco_classes[label.item()]
                vehicle_details[vehicle_type] += 1

        return {
            'total_vehicles': vehicle_count,
            'details': vehicle_details,
            'confidence_threshold': 0.5
        }

    def calculate_occupancy_rate(self, image_path, total_spots):
        """计算停车场占用率"""
        result = self.count_vehicles(image_path)
        occupancy_rate = result['total_vehicles'] / total_spots

        return {
            **result,
            'total_spots': total_spots,
            'occupancy_rate': occupancy_rate
        }

    def analyze_time_series(self, image_paths, dates, total_spots):
        """分析时间序列卫星图像"""
        results = []

        for image_path, date in zip(image_paths, dates):
            analysis = self.calculate_occupancy_rate(image_path, total_spots)
            results.append({
                'date': date,
                **analysis
            })

        df = pd.DataFrame(results)

        # 计算统计指标
        summary = {
            'avg_occupancy': df['occupancy_rate'].mean(),
            'max_occupancy': df['occupancy_rate'].max(),
            'min_occupancy': df['occupancy_rate'].min(),
            'trend': self._calculate_trend(df['occupancy_rate'])
        }

        return df, summary

    def _calculate_trend(self, series):
        """计算趋势"""
        if len(series) < 2:
            return 'insufficient_data'

        # 简单线性趋势
        x = np.arange(len(series))
        slope = np.polyfit(x, series.values, 1)[0]

        if slope > 0.01:
            return 'increasing'
        elif slope < -0.01:
            return 'decreasing'
        else:
            return 'stable'

import cv2
import numpy as np

class OilTankAnalyzer:
    """油罐库存分析"""

    def __init__(self):
        pass

    def estimate_inventory(self, image_path, tank_diameter_meters):
        """
        基于浮顶油罐阴影估计库存

        原理：浮顶油罐的顶部会随油位上下浮动，
        阴影长度与油位高度成反比
        """
        image = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)

        # 边缘检测找到油罐轮廓
        edges = cv2.Canny(image, 50, 150)

        # 霍夫圆检测找到油罐
        circles = cv2.HoughCircles(
            edges,
            cv2.HOUGH_GRADIENT,
            dp=1,
            minDist=100,
            param1=50,
            param2=30,
            minRadius=20,
            maxRadius=200
        )

        if circles is None:
            return None

        results = []
        for circle in circles[0]:
            x, y, r = circle
            tank_analysis = self._analyze_single_tank(image, x, y, r)
            results.append(tank_analysis)

        return results

    def _analyze_single_tank(self, image, x, y, r):
        """分析单个油罐"""
        # 提取油罐区域
        mask = np.zeros(image.shape, dtype=np.uint8)
        cv2.circle(mask, (int(x), int(y)), int(r), 255, -1)

        tank_region = cv2.bitwise_and(image, mask)

        # 计算亮暗区域比例（估计填充率）
        tank_pixels = tank_region[mask == 255]
        mean_brightness = np.mean(tank_pixels)

        # 亮度越高表示阴影越少，油位越高
        # 这是一个简化模型，实际需要更复杂的分析
        estimated_fill_rate = mean_brightness / 255.0

        return {
            'center': (x, y),
            'radius': r,
            'estimated_fill_rate': estimated_fill_rate
        }

import torch
import torch.nn as nn
from torchvision import models
import matplotlib.pyplot as plt
import mplfinance as mpf
from PIL import Image
import io

class KLinePatternRecognizer:
    """K线图模式识别"""

    def __init__(self, model_path=None):
        # 使用预训练的ResNet作为特征提取器
        self.model = models.resnet18(pretrained=True)
        num_features = self.model.fc.in_features

        # 替换最后一层用于分类
        self.model.fc = nn.Linear(num_features, 10)  # 10种常见形态

        if model_path:
            self.model.load_state_dict(torch.load(model_path))

        self.model.eval()

        self.pattern_names = [
            'head_and_shoulders',     # 头肩顶
            'double_top',             # 双顶
            'double_bottom',          # 双底
            'ascending_triangle',     # 上升三角形
            'descending_triangle',    # 下降三角形
            'bullish_flag',           # 上升旗形
            'bearish_flag',           # 下降旗形
            'cup_and_handle',         # 杯柄形态
            'wedge',                  # 楔形
            'channel'                 # 通道
        ]

        self.transform = transforms.Compose([
            transforms.Resize((224, 224)),
            transforms.ToTensor(),
            transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225])
        ])

    def generate_kline_image(self, df):
        """生成K线图图像"""
        # df需要包含: Open, High, Low, Close, Volume
        df.index = pd.DatetimeIndex(df['date'])

        # 创建K线图
        fig, ax = plt.subplots(figsize=(10, 6))
        mpf.plot(df, type='candle', style='yahoo', ax=ax)

        # 转换为PIL Image
        buf = io.BytesIO()
        plt.savefig(buf, format='png', dpi=100, bbox_inches='tight')
        buf.seek(0)
        image = Image.open(buf)
        plt.close()

        return image

    def recognize_pattern(self, df):
        """识别K线图形态"""
        # 生成K线图
        image = self.generate_kline_image(df)

        # 预处理
        image_tensor = self.transform(image).unsqueeze(0)

        # 预测
        with torch.no_grad():
            outputs = self.model(image_tensor)
            probs = torch.softmax(outputs, dim=1)

        # 获取预测结果
        pred_idx = torch.argmax(probs).item()
        confidence = probs[0][pred_idx].item()

        return {
            'pattern': self.pattern_names[pred_idx],
            'confidence': confidence,
            'all_probabilities': {
                name: probs[0][i].item()
                for i, name in enumerate(self.pattern_names)
            }
        }

    def detect_patterns_rolling(self, df, window_size=60, step=10):
        """滚动窗口检测形态"""
        results = []

        for i in range(0, len(df) - window_size, step):
            window_df = df.iloc[i:i+window_size].copy()
            pattern = self.recognize_pattern(window_df)

            results.append({
                'start_date': df.iloc[i]['date'],
                'end_date': df.iloc[i+window_size-1]['date'],
                **pattern
            })

        return pd.DataFrame(results)

import whisper
import librosa
import numpy as np
from transformers import pipeline

class EarningsCallAudioAnalyzer:
    """财报电话会议语音分析"""

    def __init__(self):
        # 语音识别模型
        self.whisper_model = whisper.load_model("base")

        # 情感分析模型
        self.sentiment_analyzer = pipeline(
            "sentiment-analysis",
            model="yiyanghkust/finbert-tone"
        )

    def transcribe(self, audio_path):
        """语音转文字"""
        result = self.whisper_model.transcribe(audio_path)
        return {
            'text': result['text'],
            'segments': result['segments'],
            'language': result['language']
        }

    def analyze_audio_features(self, audio_path):
        """分析语音特征"""
        # 加载音频
        y, sr = librosa.load(audio_path)

        # 提取特征
        features = {
            # 语速相关
            'tempo': librosa.beat.tempo(y=y, sr=sr)[0],

            # 音高相关
            'pitch_mean': np.mean(librosa.piptrack(y=y, sr=sr)[0]),
            'pitch_std': np.std(librosa.piptrack(y=y, sr=sr)[0]),

            # 能量相关
            'energy_mean': np.mean(librosa.feature.rms(y=y)),
            'energy_std': np.std(librosa.feature.rms(y=y)),

            # 静音比例
            'silence_ratio': self._calculate_silence_ratio(y),

            # 语音情感特征
            'mfcc': librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13).mean(axis=1).tolist()
        }

        return features

    def _calculate_silence_ratio(self, y, threshold=0.01):
        """计算静音比例"""
        silence_frames = np.sum(np.abs(y) < threshold)
        return silence_frames / len(y)

    def analyze_speaker_confidence(self, audio_path):
        """分析发言人置信度"""
        audio_features = self.analyze_audio_features(audio_path)

        # 基于语音特征估计置信度
        # 高置信度特征：语速适中、音高稳定、能量充足、少停顿
        confidence_score = 0

        # 语速评分（120-160词/分钟为理想范围）
        tempo = audio_features['tempo']
        if 120 <= tempo <= 160:
            confidence_score += 0.3
        elif 100 <= tempo <= 180:
            confidence_score += 0.15

        # 音高稳定性（标准差低表示稳定）
        if audio_features['pitch_std'] < 50:
            confidence_score += 0.25
        elif audio_features['pitch_std'] < 100:
            confidence_score += 0.1

        # 能量充足
        if audio_features['energy_mean'] > 0.1:
            confidence_score += 0.25

        # 少停顿
        if audio_features['silence_ratio'] < 0.2:
            confidence_score += 0.2
        elif audio_features['silence_ratio'] < 0.3:
            confidence_score += 0.1

        return {
            'confidence_score': confidence_score,
            'features': audio_features,
            'interpretation': self._interpret_confidence(confidence_score)
        }

    def _interpret_confidence(self, score):
        if score >= 0.7:
            return 'high_confidence'
        elif score >= 0.4:
            return 'moderate_confidence'
        else:
            return 'low_confidence'

    def full_analysis(self, audio_path):
        """完整分析"""
        # 1. 语音转文字
        transcription = self.transcribe(audio_path)

        # 2. 分析语音特征
        audio_features = self.analyze_audio_features(audio_path)

        # 3. 分析发言人置信度
        confidence = self.analyze_speaker_confidence(audio_path)

        # 4. 文本情感分析
        text = transcription['text']
        # 分段分析
        segments = text.split('.')
        sentiments = []
        for seg in segments:
            if len(seg.strip()) > 20:
                sentiment = self.sentiment_analyzer(seg[:512])
                sentiments.append(sentiment[0])

        # 汇总情感
        if sentiments:
            positive_ratio = sum(1 for s in sentiments if s['label'] == 'positive') / len(sentiments)
            negative_ratio = sum(1 for s in sentiments if s['label'] == 'negative') / len(sentiments)
        else:
            positive_ratio = negative_ratio = 0

        return {
            'transcription': transcription,
            'audio_features': audio_features,
            'speaker_confidence': confidence,
            'text_sentiment': {
                'positive_ratio': positive_ratio,
                'negative_ratio': negative_ratio,
                'total_segments': len(sentiments)
            }
        }

class CustomerServiceAnalyzer:
    """客服电话质量分析"""

    def __init__(self):
        self.whisper_model = whisper.load_model("base")
        self.sentiment_pipeline = pipeline("sentiment-analysis")

    def analyze_call(self, audio_path):
        """分析客服通话"""
        # 转录
        transcription = self.whisper_model.transcribe(audio_path)
        text = transcription['text']

        # 分析情感变化
        sentiment_trend = self._analyze_sentiment_trend(text)

        # 识别关键词
        keywords = self._extract_keywords(text)

        # 分析通话质量指标
        quality_metrics = self._calculate_quality_metrics(text, audio_path)

        return {
            'transcription': text,
            'sentiment_trend': sentiment_trend,
            'keywords': keywords,
            'quality_metrics': quality_metrics
        }

    def _analyze_sentiment_trend(self, text):
        """分析情感趋势"""
        sentences = text.split('.')
        sentiments = []

        for i, sent in enumerate(sentences):
            if len(sent.strip()) > 10:
                result = self.sentiment_pipeline(sent[:512])
                sentiments.append({
                    'position': i / len(sentences),
                    'sentiment': result[0]['label'],
                    'score': result[0]['score']
                })

        return sentiments

    def _extract_keywords(self, text):
        """提取关键词"""
        # 投诉相关
        complaint_keywords = ['问题', '不满', '投诉', '退款', '赔偿']
        # 产品相关
        product_keywords = ['理财', '基金', '保险', '贷款', '信用卡']
        # 服务相关
        service_keywords = ['办理', '查询', '开户', '密码', '转账']

        found_keywords = {
            'complaints': [kw for kw in complaint_keywords if kw in text],
            'products': [kw for kw in product_keywords if kw in text],
            'services': [kw for kw in service_keywords if kw in text]
        }

        return found_keywords

    def _calculate_quality_metrics(self, text, audio_path):
        """计算通话质量指标"""
        y, sr = librosa.load(audio_path)
        duration = len(y) / sr

        # 通话时长
        # 礼貌用语检测
        polite_words = ['您好', '谢谢', '请', '抱歉', '感谢']
        polite_count = sum(text.count(word) for word in polite_words)

        # 问题解决标志词
        resolution_words = ['已处理', '已解决', '帮您', '可以']
        resolution_count = sum(text.count(word) for word in resolution_words)

        return {
            'duration_seconds': duration,
            'politeness_score': min(polite_count / 5, 1.0),
            'resolution_indicators': resolution_count,
            'words_per_minute': len(text) / (duration / 60)
        }

import networkx as nx
from py2neo import Graph, Node, Relationship
import json

class FinancialKnowledgeGraph:
    """金融知识图谱"""

    def __init__(self, neo4j_uri=None, neo4j_user=None, neo4j_password=None):
        if neo4j_uri:
            self.graph_db = Graph(neo4j_uri, auth=(neo4j_user, neo4j_password))
        else:
            self.graph_db = None

        # 内存图
        self.nx_graph = nx.DiGraph()

        # 实体类型
        self.entity_types = [
            'Company', 'Person', 'Product', 'Industry',
            'Location', 'Event', 'Document'
        ]

        # 关系类型
        self.relation_types = [
            'INVEST_IN', 'SUBSIDIARY_OF', 'COMPETE_WITH',
            'SUPPLY_TO', 'PARTNER_WITH', 'LOCATED_IN',
            'WORK_FOR', 'PRODUCE', 'BELONG_TO'
        ]

    def add_entity(self, entity_id, entity_type, properties):
        """添加实体"""
        self.nx_graph.add_node(
            entity_id,
            type=entity_type,
            **properties
        )

    def add_relation(self, source_id, target_id, relation_type, properties=None):
        """添加关系"""
        self.nx_graph.add_edge(
            source_id,
            target_id,
            type=relation_type,
            **(properties or {})
        )

    def find_shortest_path(self, source_id, target_id):
        """查找最短路径"""
        try:
            path = nx.shortest_path(self.nx_graph, source_id, target_id)
            return path
        except nx.NetworkXNoPath:
            return None

    def find_all_paths(self, source_id, target_id, max_depth=5):
        """查找所有路径"""
        try:
            paths = list(nx.all_simple_paths(
                self.nx_graph, source_id, target_id, cutoff=max_depth
            ))
            return paths
        except nx.NetworkXNoPath:
            return []

    def get_neighbors(self, entity_id, depth=1):
        """获取邻居节点"""
        neighbors = set()
        current_level = {entity_id}

        for _ in range(depth):
            next_level = set()
            for node in current_level:
                successors = set(self.nx_graph.successors(node))
                predecessors = set(self.nx_graph.predecessors(node))
                next_level.update(successors)
                next_level.update(predecessors)
            neighbors.update(next_level)
            current_level = next_level - neighbors

        neighbors.discard(entity_id)
        return neighbors

    def calculate_centrality(self):
        """计算节点中心度"""
        return {
            'degree': nx.degree_centrality(self.nx_graph),
            'betweenness': nx.betweenness_centrality(self.nx_graph),
            'pagerank': nx.pagerank(self.nx_graph)
        }

    def find_communities(self):
        """发现社区"""
        # 转换为无向图进行社区检测
        undirected = self.nx_graph.to_undirected()
        communities = list(nx.community.greedy_modularity_communities(undirected))
        return [list(c) for c in communities]


class RiskPropagationAnalyzer:
    """风险传染分析"""

    def __init__(self, knowledge_graph):
        self.kg = knowledge_graph

    def analyze_risk_exposure(self, company_id, risk_source_id):
        """分析风险暴露"""
        # 查找所有连接路径
        paths = self.kg.find_all_paths(risk_source_id, company_id, max_depth=4)

        if not paths:
            return {
                'exposure': 'none',
                'paths': [],
                'risk_score': 0
            }

        # 计算风险分数
        risk_score = self._calculate_risk_score(paths)

        return {
            'exposure': self._classify_exposure(risk_score),
            'paths': paths,
            'risk_score': risk_score,
            'num_paths': len(paths),
            'min_distance': min(len(p) for p in paths)
        }

    def _calculate_risk_score(self, paths):
        """计算风险分数"""
        # 路径越短、路径越多，风险越大
        if not paths:
            return 0

        min_length = min(len(p) for p in paths)
        num_paths = len(paths)

        # 距离衰减
        distance_factor = 1 / min_length

        # 路径数量加权
        path_factor = min(num_paths / 10, 1)

        return distance_factor * 0.7 + path_factor * 0.3

    def _classify_exposure(self, score):
        if score > 0.7:
            return 'high'
        elif score > 0.3:
            return 'medium'
        elif score > 0:
            return 'low'
        else:
            return 'none'

    def find_risk_clusters(self, risk_threshold=0.5):
        """发现风险集群"""
        centrality = self.kg.calculate_centrality()
        pagerank = centrality['pagerank']

        # 高中心度节点可能是风险传播的关键节点
        high_risk_nodes = [
            node for node, score in pagerank.items()
            if score > risk_threshold * max(pagerank.values())
        ]

        return high_risk_nodes

    def simulate_risk_propagation(self, initial_node, infection_rate=0.3, max_steps=10):
        """模拟风险传播"""
        infected = {initial_node}
        history = [{initial_node}]

        for step in range(max_steps):
            new_infected = set()
            for node in infected:
                neighbors = self.kg.get_neighbors(node, depth=1)
                for neighbor in neighbors:
                    if neighbor not in infected and np.random.random() < infection_rate:
                        new_infected.add(neighbor)

            if not new_infected:
                break

            infected.update(new_infected)
            history.append(new_infected)

        return {
            'total_infected': len(infected),
            'infection_history': history,
            'steps': len(history)
        }

class CompanyRelationGraph:
    """企业关联图谱"""

    def __init__(self):
        self.kg = FinancialKnowledgeGraph()

    def build_from_data(self, companies_df, relations_df):
        """从数据构建图谱"""
        # 添加公司节点
        for _, row in companies_df.iterrows():
            self.kg.add_entity(
                entity_id=row['company_id'],
                entity_type='Company',
                properties={
                    'name': row['name'],
                    'industry': row['industry'],
                    'market_cap': row.get('market_cap'),
                    'revenue': row.get('revenue')
                }
            )

        # 添加关系
        for _, row in relations_df.iterrows():
            self.kg.add_relation(
                source_id=row['source_company'],
                target_id=row['target_company'],
                relation_type=row['relation_type'],
                properties={'weight': row.get('weight', 1.0)}
            )

    def find_supply_chain(self, company_id):
        """查找供应链"""
        suppliers = []
        customers = []

        # 向上游查找供应商
        for predecessor in self.kg.nx_graph.predecessors(company_id):
            edge_data = self.kg.nx_graph.get_edge_data(predecessor, company_id)
            if edge_data.get('type') == 'SUPPLY_TO':
                suppliers.append(predecessor)

        # 向下游查找客户
        for successor in self.kg.nx_graph.successors(company_id):
            edge_data = self.kg.nx_graph.get_edge_data(company_id, successor)
            if edge_data.get('type') == 'SUPPLY_TO':
                customers.append(successor)

        return {
            'suppliers': suppliers,
            'customers': customers,
            'supply_chain_depth': self._calculate_chain_depth(company_id)
        }

    def _calculate_chain_depth(self, company_id):
        """计算供应链深度"""
        # 向上游追溯
        upstream_depth = 0
        current = {company_id}
        visited = set()

        while current:
            next_level = set()
            for node in current:
                if node in visited:
                    continue
                visited.add(node)
                for pred in self.kg.nx_graph.predecessors(node):
                    edge = self.kg.nx_graph.get_edge_data(pred, node)
                    if edge.get('type') == 'SUPPLY_TO':
                        next_level.add(pred)
            if next_level:
                upstream_depth += 1
            current = next_level

        return upstream_depth

    def find_competitors(self, company_id):
        """查找竞争对手"""
        competitors = []

        # 直接竞争关系
        for neighbor in self.kg.get_neighbors(company_id, depth=1):
            edge1 = self.kg.nx_graph.get_edge_data(company_id, neighbor)
            edge2 = self.kg.nx_graph.get_edge_data(neighbor, company_id)

            if (edge1 and edge1.get('type') == 'COMPETE_WITH') or \
               (edge2 and edge2.get('type') == 'COMPETE_WITH'):
                competitors.append(neighbor)

        # 同行业公司
        company_data = self.kg.nx_graph.nodes.get(company_id, {})
        industry = company_data.get('industry')

        if industry:
            for node, data in self.kg.nx_graph.nodes(data=True):
                if data.get('industry') == industry and node != company_id:
                    if node not in competitors:
                        competitors.append(node)

        return competitors

    def analyze_group_structure(self, company_id):
        """分析集团架构"""
        # 查找控股关系
        subsidiaries = []
        parent = None

        for pred in self.kg.nx_graph.predecessors(company_id):
            edge = self.kg.nx_graph.get_edge_data(pred, company_id)
            if edge.get('type') == 'SUBSIDIARY_OF':
                parent = pred
                break

        for succ in self.kg.nx_graph.successors(company_id):
            edge = self.kg.nx_graph.get_edge_data(company_id, succ)
            if edge.get('type') in ['SUBSIDIARY_OF', 'INVEST_IN']:
                weight = edge.get('weight', 0)
                if weight > 0.5:  # 控股比例>50%
                    subsidiaries.append({
                        'company': succ,
                        'ownership': weight
                    })

        return {
            'parent': parent,
            'subsidiaries': subsidiaries
        }

import torch
import torch.nn as nn

class EarlyFusionModel(nn.Module):
    """早期融合模型"""

    def __init__(self, text_dim, image_dim, audio_dim, num_classes):
        super().__init__()

        # 特征提取器
        self.text_encoder = nn.Sequential(
            nn.Linear(text_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3)
        )

        self.image_encoder = nn.Sequential(
            nn.Linear(image_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3)
        )

        self.audio_encoder = nn.Sequential(
            nn.Linear(audio_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3)
        )

        # 融合层
        self.fusion = nn.Sequential(
            nn.Linear(256 * 3, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, num_classes)
        )

    def forward(self, text_feat, image_feat, audio_feat):
        # 编码各模态
        text_encoded = self.text_encoder(text_feat)
        image_encoded = self.image_encoder(image_feat)
        audio_encoded = self.audio_encoder(audio_feat)

        # 拼接融合
        fused = torch.cat([text_encoded, image_encoded, audio_encoded], dim=1)

        # 分类
        output = self.fusion(fused)
        return output

class LateFusionModel(nn.Module):
    """晚期融合模型"""

    def __init__(self, text_dim, image_dim, audio_dim, num_classes):
        super().__init__()

        # 各模态独立分类器
        self.text_classifier = nn.Sequential(
            nn.Linear(text_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, num_classes)
        )

        self.image_classifier = nn.Sequential(
            nn.Linear(image_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, num_classes)
        )

        self.audio_classifier = nn.Sequential(
            nn.Linear(audio_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, num_classes)
        )

        # 融合权重（可学习）
        self.fusion_weights = nn.Parameter(torch.ones(3) / 3)

    def forward(self, text_feat, image_feat, audio_feat):
        # 各模态预测
        text_logits = self.text_classifier(text_feat)
        image_logits = self.image_classifier(image_feat)
        audio_logits = self.audio_classifier(audio_feat)

        # 加权融合
        weights = torch.softmax(self.fusion_weights, dim=0)
        fused_logits = (
            weights[0] * text_logits +
            weights[1] * image_logits +
            weights[2] * audio_logits
        )

        return fused_logits

class AttentionFusionModel(nn.Module):
    """注意力融合模型"""

    def __init__(self, text_dim, image_dim, audio_dim, hidden_dim, num_classes):
        super().__init__()

        # 投影到相同维度
        self.text_proj = nn.Linear(text_dim, hidden_dim)
        self.image_proj = nn.Linear(image_dim, hidden_dim)
        self.audio_proj = nn.Linear(audio_dim, hidden_dim)

        # 多头注意力
        self.multihead_attn = nn.MultiheadAttention(
            embed_dim=hidden_dim,
            num_heads=8,
            dropout=0.1
        )

        # 分类器
        self.classifier = nn.Sequential(
            nn.Linear(hidden_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, num_classes)
        )

    def forward(self, text_feat, image_feat, audio_feat):
        # 投影
        text_proj = self.text_proj(text_feat).unsqueeze(0)  # [1, batch, hidden]
        image_proj = self.image_proj(image_feat).unsqueeze(0)
        audio_proj = self.audio_proj(audio_feat).unsqueeze(0)

        # 拼接为序列
        sequence = torch.cat([text_proj, image_proj, audio_proj], dim=0)  # [3, batch, hidden]

        # 自注意力
        attended, attn_weights = self.multihead_attn(
            sequence, sequence, sequence
        )

        # 全局池化
        pooled = attended.mean(dim=0)  # [batch, hidden]

        # 分类
        output = self.classifier(pooled)

        return output, attn_weights

class CEOToneResearch:
    """CEO语气研究"""

    def __init__(self):
        self.audio_analyzer = EarningsCallAudioAnalyzer()

    def analyze_ceo_segment(self, audio_path, start_time, end_time):
        """分析CEO发言段落"""
        # 提取CEO发言音频
        y, sr = librosa.load(audio_path, offset=start_time, duration=end_time-start_time)

        # 分析语音特征
        features = {
            'pitch_variability': np.std(librosa.piptrack(y=y, sr=sr)[0]),
            'speaking_rate': self._estimate_speaking_rate(y, sr),
            'pause_frequency': self._count_pauses(y, sr),
            'energy_level': np.mean(librosa.feature.rms(y=y))
        }

        return features

    def _estimate_speaking_rate(self, y, sr):
        """估计语速"""
        # 基于能量变化估计
        rms = librosa.feature.rms(y=y)
        # 计算变化频率
        diff = np.diff(rms[0])
        zero_crossings = np.sum(np.diff(np.sign(diff)) != 0)
        duration = len(y) / sr
        return zero_crossings / duration * 60  # 每分钟变化次数

    def _count_pauses(self, y, sr, threshold=0.02, min_pause_duration=0.3):
        """统计停顿"""
        rms = librosa.feature.rms(y=y)[0]
        pause_frames = rms < threshold

        # 连续静音帧
        pauses = 0
        pause_length = 0
        for is_pause in pause_frames:
            if is_pause:
                pause_length += 1
            else:
                if pause_length * 512 / sr > min_pause_duration:
                    pauses += 1
                pause_length = 0

        return pauses

    def create_tone_factor(self, calls_data):
        """构建语气因子"""
        results = []

        for call in calls_data:
            features = self.analyze_ceo_segment(
                call['audio_path'],
                call['ceo_start'],
                call['ceo_end']
            )

            # 构建综合语气得分
            tone_score = (
                -0.3 * features['pitch_variability'] / 100 +  # 音高稳定性
                0.3 * features['speaking_rate'] / 150 +        # 语速适中
                -0.2 * features['pause_frequency'] / 10 +     # 少停顿
                0.2 * features['energy_level'] * 10            # 能量充足
            )

            results.append({
                'date': call['date'],
                'stock_code': call['stock_code'],
                'tone_score': tone_score,
                **features
            })

        return pd.DataFrame(results)

class RetailSalesPredictor:
    """零售销售预测"""

    def __init__(self):
        self.satellite_analyzer = SatelliteImageAnalyzer()

    def build_parking_dataset(self, image_data, sales_data):
        """构建停车场数据集"""
        results = []

        for img_info in image_data:
            # 分析卫星图
            analysis = self.satellite_analyzer.calculate_occupancy_rate(
                img_info['image_path'],
                img_info['total_spots']
            )

            # 关联销售数据
            sales_match = sales_data[
                (sales_data['store_id'] == img_info['store_id']) &
                (sales_data['date'] == img_info['date'])
            ]

            if len(sales_match) > 0:
                results.append({
                    'date': img_info['date'],
                    'store_id': img_info['store_id'],
                    'occupancy_rate': analysis['occupancy_rate'],
                    'vehicle_count': analysis['total_vehicles'],
                    'actual_sales': sales_match['sales'].iloc[0]
                })

        return pd.DataFrame(results)

    def train_prediction_model(self, df):
        """训练预测模型"""
        from sklearn.ensemble import RandomForestRegressor
        from sklearn.model_selection import train_test_split

        X = df[['occupancy_rate', 'vehicle_count']]
        y = df['actual_sales']

        X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

        model = RandomForestRegressor(n_estimators=100)
        model.fit(X_train, y_train)

        # 评估
        train_score = model.score(X_train, y_train)
        test_score = model.score(X_test, y_test)

        return {
            'model': model,
            'train_r2': train_score,
            'test_r2': test_score
        }

    def calculate_predictive_alpha(self, predictions_df, actual_sales_df, market_returns):
        """计算预测Alpha"""
        # 基于预测构建交易信号
        predictions_df['signal'] = predictions_df['predicted_sales'].pct_change()

        # 计算策略收益
        merged = predictions_df.merge(market_returns, on='date')
        merged['strategy_return'] = merged['signal'].shift(1) * merged['stock_return']

        # 计算Alpha（相对于市场的超额收益）
        alpha = merged['strategy_return'].mean() - merged['market_return'].mean()

        return {
            'alpha': alpha * 252,  # 年化
            'sharpe': merged['strategy_return'].mean() / merged['strategy_return'].std() * np.sqrt(252),
            'information_ratio': alpha / merged['strategy_return'].std() * np.sqrt(252)
        }
