import requests
from bs4 import BeautifulSoup
import time
import random

class FinancialCrawler:
    """金融数据爬虫基类"""

    def __init__(self):
        self.session = requests.Session()
        self.headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
            'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9',
            'Accept-Language': 'zh-CN,zh;q=0.9,en;q=0.8'
        }

    def get_page(self, url, retry=3):
        """获取页面内容"""
        for i in range(retry):
            try:
                response = self.session.get(
                    url,
                    headers=self.headers,
                    timeout=10
                )
                response.raise_for_status()
                response.encoding = 'utf-8'
                return response.text
            except Exception as e:
                print(f"请求失败 ({i+1}/{retry}): {e}")
                time.sleep(random.uniform(1, 3))
        return None

    def parse_table(self, html, table_id=None):
        """解析HTML表格"""
        soup = BeautifulSoup(html, 'html.parser')
        if table_id:
            table = soup.find('table', id=table_id)
        else:
            table = soup.find('table')

        if not table:
            return None

        rows = []
        for tr in table.find_all('tr'):
            cells = [td.get_text(strip=True) for td in tr.find_all(['td', 'th'])]
            if cells:
                rows.append(cells)

        return pd.DataFrame(rows[1:], columns=rows[0]) if rows else None

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.options import Options

class DynamicCrawler:
    """动态页面爬虫"""

    def __init__(self, headless=True):
        options = Options()
        if headless:
            options.add_argument('--headless')
        options.add_argument('--disable-gpu')
        options.add_argument('--no-sandbox')

        self.driver = webdriver.Chrome(options=options)
        self.wait = WebDriverWait(self.driver, 10)

    def get_dynamic_content(self, url, wait_element=None):
        """获取动态加载的内容"""
        self.driver.get(url)

        if wait_element:
            self.wait.until(
                EC.presence_of_element_located((By.CSS_SELECTOR, wait_element))
            )

        return self.driver.page_source

    def scroll_to_load(self, scroll_times=5, wait_time=2):
        """滚动加载更多内容"""
        for _ in range(scroll_times):
            self.driver.execute_script(
                "window.scrollTo(0, document.body.scrollHeight);"
            )
            time.sleep(wait_time)

        return self.driver.page_source

    def close(self):
        """关闭浏览器"""
        self.driver.quit()

数据清洗流程

python
import pandas as pd
import numpy as np
from scipy import stats

class DataCleaner:
    """数据清洗工具"""

    def __init__(self, df):
        self.df = df.copy()
        self.cleaning_log = []

    def handle_missing_values(self, strategy='auto', columns=None):
        """处理缺失值"""
        if columns is None:
            columns = self.df.columns

        for col in columns:
            missing_count = self.df[col].isna().sum()
            if missing_count == 0:
                continue

            missing_ratio = missing_count / len(self.df)

            if strategy == 'auto':
                if missing_ratio > 0.5:
                    # 缺失超过50%，删除列
                    self.df.drop(col, axis=1, inplace=True)
                    self.cleaning_log.append(f"删除列 {col}：缺失率 {missing_ratio:.1%}")
                elif self.df[col].dtype in ['float64', 'int64']:
                    # 数值型：中位数填充
                    median_val = self.df[col].median()
                    self.df[col].fillna(median_val, inplace=True)
                    self.cleaning_log.append(f"列 {col} 用中位数 {median_val} 填充")
                else:
                    # 类别型：众数填充
                    mode_val = self.df[col].mode()[0]
                    self.df[col].fillna(mode_val, inplace=True)
                    self.cleaning_log.append(f"列 {col} 用众数 '{mode_val}' 填充")

            elif strategy == 'drop':
                self.df.dropna(subset=[col], inplace=True)

            elif strategy == 'ffill':
                self.df[col].fillna(method='ffill', inplace=True)

            elif strategy == 'interpolate':
                self.df[col].interpolate(method='linear', inplace=True)

        return self

    def handle_outliers(self, columns, method='zscore', threshold=3):
        """处理异常值"""
        for col in columns:
            if self.df[col].dtype not in ['float64', 'int64']:
                continue

            if method == 'zscore':
                z_scores = np.abs(stats.zscore(self.df[col].dropna()))
                outlier_mask = z_scores > threshold
                outlier_count = outlier_mask.sum()

                # 用边界值替换
                mean = self.df[col].mean()
                std = self.df[col].std()
                lower, upper = mean - threshold * std, mean + threshold * std
                self.df[col] = self.df[col].clip(lower, upper)

            elif method == 'iqr':
                Q1 = self.df[col].quantile(0.25)
                Q3 = self.df[col].quantile(0.75)
                IQR = Q3 - Q1
                lower = Q1 - 1.5 * IQR
                upper = Q3 + 1.5 * IQR

                outlier_count = ((self.df[col] < lower) | (self.df[col] > upper)).sum()
                self.df[col] = self.df[col].clip(lower, upper)

            elif method == 'percentile':
                lower = self.df[col].quantile(0.01)
                upper = self.df[col].quantile(0.99)
                self.df[col] = self.df[col].clip(lower, upper)

            self.cleaning_log.append(
                f"列 {col} 处理 {outlier_count} 个异常值 (方法: {method})"
            )

        return self

    def remove_duplicates(self, subset=None, keep='first'):
        """删除重复数据"""
        before_count = len(self.df)
        self.df.drop_duplicates(subset=subset, keep=keep, inplace=True)
        after_count = len(self.df)

        self.cleaning_log.append(
            f"删除 {before_count - after_count} 条重复数据"
        )
        return self

    def standardize_columns(self, column_mapping):
        """标准化列名"""
        self.df.rename(columns=column_mapping, inplace=True)
        return self

    def convert_types(self, type_mapping):
        """转换数据类型"""
        for col, dtype in type_mapping.items():
            try:
                if dtype == 'datetime':
                    self.df[col] = pd.to_datetime(self.df[col])
                elif dtype == 'numeric':
                    self.df[col] = pd.to_numeric(self.df[col], errors='coerce')
                else:
                    self.df[col] = self.df[col].astype(dtype)
            except Exception as e:
                self.cleaning_log.append(f"列 {col} 类型转换失败: {e}")
        return self

    def get_result(self):
        """获取清洗结果"""
        return self.df, self.cleaning_log

class StockDataProcessor:
    """股票数据专用处理器"""

    def __init__(self, df):
        self.df = df.copy()

    def adjust_price(self, method='forward'):
        """复权处理"""
        if method == 'forward':  # 前复权
            # 假设df中有adj_factor（复权因子）列
            if 'adj_factor' in self.df.columns:
                latest_factor = self.df['adj_factor'].iloc[-1]
                for col in ['open', 'high', 'low', 'close']:
                    self.df[f'{col}_adj'] = self.df[col] * self.df['adj_factor'] / latest_factor

        elif method == 'backward':  # 后复权
            if 'adj_factor' in self.df.columns:
                first_factor = self.df['adj_factor'].iloc[0]
                for col in ['open', 'high', 'low', 'close']:
                    self.df[f'{col}_adj'] = self.df[col] * self.df['adj_factor'] / first_factor

        return self

    def handle_suspension(self, method='ffill'):
        """处理停牌数据"""
        # 识别停牌日（成交量为0或NaN）
        suspension_mask = (self.df['volume'] == 0) | self.df['volume'].isna()

        if method == 'ffill':
            # 用前一天数据填充
            self.df.loc[suspension_mask, ['open', 'high', 'low', 'close']] = np.nan
            self.df[['open', 'high', 'low', 'close']] = \
                self.df[['open', 'high', 'low', 'close']].fillna(method='ffill')

        elif method == 'remove':
            # 删除停牌日
            self.df = self.df[~suspension_mask]

        return self

    def align_trading_days(self, trading_calendar):
        """对齐交易日"""
        self.df['date'] = pd.to_datetime(self.df['date'])
        trading_calendar = pd.to_datetime(trading_calendar)

        # 只保留交易日数据
        self.df = self.df[self.df['date'].isin(trading_calendar)]

        return self

    def calculate_returns(self, periods=[1, 5, 20]):
        """计算收益率"""
        for period in periods:
            self.df[f'return_{period}d'] = \
                self.df['close'].pct_change(period)

            # 对数收益率
            self.df[f'log_return_{period}d'] = \
                np.log(self.df['close'] / self.df['close'].shift(period))

        return self

class PriceVolumeFeatures:
    """价量特征工程"""

    def __init__(self, df):
        self.df = df.copy()

    def add_moving_averages(self, windows=[5, 10, 20, 60]):
        """添加移动平均线"""
        for window in windows:
            self.df[f'ma_{window}'] = self.df['close'].rolling(window).mean()
            self.df[f'ma_{window}_ratio'] = self.df['close'] / self.df[f'ma_{window}']

            # 成交量均线
            self.df[f'vol_ma_{window}'] = self.df['volume'].rolling(window).mean()

        return self

    def add_volatility(self, windows=[5, 20, 60]):
        """添加波动率指标"""
        returns = self.df['close'].pct_change()

        for window in windows:
            # 历史波动率
            self.df[f'volatility_{window}'] = returns.rolling(window).std() * np.sqrt(252)

            # 真实波动幅度（ATR）
            high_low = self.df['high'] - self.df['low']
            high_close = abs(self.df['high'] - self.df['close'].shift(1))
            low_close = abs(self.df['low'] - self.df['close'].shift(1))
            true_range = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
            self.df[f'atr_{window}'] = true_range.rolling(window).mean()

        return self

    def add_momentum(self, windows=[5, 10, 20]):
        """添加动量指标"""
        for window in windows:
            # 动量
            self.df[f'momentum_{window}'] = self.df['close'].pct_change(window)

            # RSI
            delta = self.df['close'].diff()
            gain = (delta.where(delta > 0, 0)).rolling(window).mean()
            loss = (-delta.where(delta < 0, 0)).rolling(window).mean()
            rs = gain / loss
            self.df[f'rsi_{window}'] = 100 - (100 / (1 + rs))

        return self

    def add_volume_features(self):
        """添加成交量特征"""
        # 量比
        self.df['volume_ratio'] = self.df['volume'] / self.df['volume'].rolling(5).mean()

        # 换手率变化
        if 'turnover_rate' in self.df.columns:
            self.df['turnover_change'] = self.df['turnover_rate'].pct_change()

        # 量价关系
        self.df['price_volume_corr'] = \
            self.df['close'].rolling(20).corr(self.df['volume'])

        return self

    def add_technical_indicators(self):
        """添加技术指标"""
        # MACD
        exp1 = self.df['close'].ewm(span=12, adjust=False).mean()
        exp2 = self.df['close'].ewm(span=26, adjust=False).mean()
        self.df['macd'] = exp1 - exp2
        self.df['macd_signal'] = self.df['macd'].ewm(span=9, adjust=False).mean()
        self.df['macd_hist'] = self.df['macd'] - self.df['macd_signal']

        # 布林带
        self.df['bb_middle'] = self.df['close'].rolling(20).mean()
        self.df['bb_std'] = self.df['close'].rolling(20).std()
        self.df['bb_upper'] = self.df['bb_middle'] + 2 * self.df['bb_std']
        self.df['bb_lower'] = self.df['bb_middle'] - 2 * self.df['bb_std']
        self.df['bb_width'] = (self.df['bb_upper'] - self.df['bb_lower']) / self.df['bb_middle']
        self.df['bb_position'] = (self.df['close'] - self.df['bb_lower']) / \
                                  (self.df['bb_upper'] - self.df['bb_lower'])

        # KDJ
        low_min = self.df['low'].rolling(9).min()
        high_max = self.df['high'].rolling(9).max()
        rsv = (self.df['close'] - low_min) / (high_max - low_min) * 100
        self.df['kdj_k'] = rsv.ewm(com=2, adjust=False).mean()
        self.df['kdj_d'] = self.df['kdj_k'].ewm(com=2, adjust=False).mean()
        self.df['kdj_j'] = 3 * self.df['kdj_k'] - 2 * self.df['kdj_d']

        return self

    def get_features(self):
        """获取所有特征"""
        return self.df

class FinancialFeatures:
    """财务特征工程"""

    def __init__(self, df):
        self.df = df.copy()

    def add_profitability_ratios(self):
        """添加盈利能力指标"""
        # 毛利率
        self.df['gross_margin'] = \
            (self.df['revenue'] - self.df['cost']) / self.df['revenue']

        # 净利率
        self.df['net_margin'] = self.df['net_income'] / self.df['revenue']

        # ROE（净资产收益率）
        self.df['roe'] = self.df['net_income'] / self.df['equity']

        # ROA（总资产收益率）
        self.df['roa'] = self.df['net_income'] / self.df['total_assets']

        # ROIC（投入资本回报率）
        invested_capital = self.df['equity'] + self.df['long_term_debt']
        self.df['roic'] = self.df['operating_income'] * (1 - 0.25) / invested_capital

        return self

    def add_solvency_ratios(self):
        """添加偿债能力指标"""
        # 资产负债率
        self.df['debt_ratio'] = self.df['total_liabilities'] / self.df['total_assets']

        # 流动比率
        self.df['current_ratio'] = self.df['current_assets'] / self.df['current_liabilities']

        # 速动比率
        self.df['quick_ratio'] = \
            (self.df['current_assets'] - self.df['inventory']) / self.df['current_liabilities']

        # 利息覆盖倍数
        self.df['interest_coverage'] = \
            self.df['operating_income'] / self.df['interest_expense']

        return self

    def add_efficiency_ratios(self):
        """添加运营效率指标"""
        # 应收账款周转率
        self.df['receivable_turnover'] = self.df['revenue'] / self.df['accounts_receivable']

        # 存货周转率
        self.df['inventory_turnover'] = self.df['cost'] / self.df['inventory']

        # 总资产周转率
        self.df['asset_turnover'] = self.df['revenue'] / self.df['total_assets']

        # 应收账款周转天数
        self.df['receivable_days'] = 365 / self.df['receivable_turnover']

        # 存货周转天数
        self.df['inventory_days'] = 365 / self.df['inventory_turnover']

        return self

    def add_growth_ratios(self):
        """添加成长能力指标"""
        # 营收增长率
        self.df['revenue_growth'] = self.df['revenue'].pct_change()

        # 净利润增长率
        self.df['income_growth'] = self.df['net_income'].pct_change()

        # 总资产增长率
        self.df['asset_growth'] = self.df['total_assets'].pct_change()

        # 可持续增长率
        self.df['sustainable_growth'] = \
            self.df['roe'] * (1 - self.df['dividend_payout_ratio'])

        return self

    def add_valuation_ratios(self, market_cap=None):
        """添加估值指标"""
        if market_cap is not None:
            self.df['market_cap'] = market_cap

        if 'market_cap' in self.df.columns:
            # 市盈率
            self.df['pe_ratio'] = self.df['market_cap'] / self.df['net_income']

            # 市净率
            self.df['pb_ratio'] = self.df['market_cap'] / self.df['equity']

            # 市销率
            self.df['ps_ratio'] = self.df['market_cap'] / self.df['revenue']

            # EV/EBITDA
            enterprise_value = self.df['market_cap'] + self.df['total_debt'] - self.df['cash']
            self.df['ev_ebitda'] = enterprise_value / self.df['ebitda']

        return self

    def add_quality_indicators(self):
        """添加质量指标"""
        # 盈利质量（经营现金流/净利润）
        self.df['earnings_quality'] = \
            self.df['operating_cashflow'] / self.df['net_income']

        # 资本支出比率
        self.df['capex_ratio'] = self.df['capex'] / self.df['operating_cashflow']

        # 自由现金流
        self.df['free_cashflow'] = self.df['operating_cashflow'] - self.df['capex']

        # 应计项目占比
        self.df['accruals'] = (self.df['net_income'] - self.df['operating_cashflow']) \
                               / self.df['total_assets']

        return self

因子合成与标准化

python
class FactorProcessor:
    """因子处理器"""

    def __init__(self, df):
        self.df = df.copy()

    def standardize(self, columns, method='zscore'):
        """因子标准化"""
        for col in columns:
            if method == 'zscore':
                # Z-score标准化
                mean = self.df[col].mean()
                std = self.df[col].std()
                self.df[f'{col}_std'] = (self.df[col] - mean) / std

            elif method == 'rank':
                # 排序百分位
                self.df[f'{col}_std'] = self.df[col].rank(pct=True)

            elif method == 'minmax':
                # Min-Max归一化
                min_val = self.df[col].min()
                max_val = self.df[col].max()
                self.df[f'{col}_std'] = (self.df[col] - min_val) / (max_val - min_val)

        return self

    def neutralize(self, factor_col, neutralize_cols):
        """因子中性化（行业、市值中性）"""
        import statsmodels.api as sm

        X = self.df[neutralize_cols]
        X = sm.add_constant(X)
        y = self.df[factor_col]

        model = sm.OLS(y, X, missing='drop').fit()
        self.df[f'{factor_col}_neutral'] = model.resid

        return self

    def winsorize(self, columns, limits=(0.01, 0.99)):
        """缩尾处理"""
        from scipy.stats import mstats

        for col in columns:
            self.df[col] = mstats.winsorize(
                self.df[col],
                limits=[limits[0], 1 - limits[1]]
            )

        return self

    def combine_factors(self, factor_cols, weights=None, method='equal'):
        """因子合成"""
        if method == 'equal':
            weights = [1 / len(factor_cols)] * len(factor_cols)
        elif method == 'ic_weighted':
            # 使用IC加权（需要收益率数据）
            weights = self._calculate_ic_weights(factor_cols)

        self.df['combined_factor'] = sum(
            w * self.df[col] for w, col in zip(weights, factor_cols)
        )

        return self

    def _calculate_ic_weights(self, factor_cols, return_col='future_return'):
        """计算IC加权权重"""
        ics = []
        for col in factor_cols:
            ic = self.df[col].corr(self.df[return_col])
            ics.append(abs(ic))

        total_ic = sum(ics)
        return [ic / total_ic for ic in ics]

from sqlalchemy import create_engine, text
from sqlalchemy.orm import sessionmaker
import pandas as pd

class FinancialDatabase:
    """金融数据库操作类"""

    def __init__(self, connection_string):
        self.engine = create_engine(connection_string, pool_size=10)
        self.Session = sessionmaker(bind=self.engine)

    def save_dataframe(self, df, table_name, if_exists='append', chunksize=10000):
        """保存DataFrame到数据库"""
        df.to_sql(
            table_name,
            self.engine,
            if_exists=if_exists,
            index=False,
            chunksize=chunksize,
            method='multi'
        )

    def query(self, sql, params=None):
        """执行SQL查询"""
        with self.engine.connect() as conn:
            result = pd.read_sql(text(sql), conn, params=params)
        return result

    def get_stock_price(self, stock_code, start_date, end_date):
        """获取股票价格数据"""
        sql = """
            SELECT stock_code, trade_date, open_price, high_price,
                   low_price, close_price, volume, amount
            FROM daily_price
            WHERE stock_code = :code
              AND trade_date BETWEEN :start AND :end
            ORDER BY trade_date
        """
        return self.query(sql, {
            'code': stock_code,
            'start': start_date,
            'end': end_date
        })

    def get_latest_factors(self, factor_names, trade_date):
        """获取最新因子值"""
        placeholders = ','.join([':f' + str(i) for i in range(len(factor_names))])
        sql = f"""
            SELECT stock_code, factor_name, factor_value
            FROM factor_values
            WHERE factor_name IN ({placeholders})
              AND trade_date = :date
        """
        params = {'date': trade_date}
        params.update({f'f{i}': name for i, name in enumerate(factor_names)})

        df = self.query(sql, params)
        return df.pivot(index='stock_code', columns='factor_name', values='factor_value')

    def upsert_factors(self, df):
        """更新或插入因子数据"""
        # 使用INSERT ... ON DUPLICATE KEY UPDATE
        sql = """
            INSERT INTO factor_values (stock_code, trade_date, factor_name, factor_value)
            VALUES (:code, :date, :name, :value)
            ON DUPLICATE KEY UPDATE factor_value = :value
        """

        with self.engine.connect() as conn:
            for _, row in df.iterrows():
                conn.execute(text(sql), {
                    'code': row['stock_code'],
                    'date': row['trade_date'],
                    'name': row['factor_name'],
                    'value': row['factor_value']
                })
            conn.commit()

            '1h': 'toStartOfHour(tick_time)',
            '1d': 'toDate(tick_time)'

                argMax(price, tick_time) as close,
                sum(volume) as volume,
                sum(amount) as amount

            WHERE stock_code = %(code)s
              AND tick_time BETWEEN %(start)s AND %(end)s
            GROUP BY time
            ORDER BY time

            'end': end_time
        })

核心公式

1

收益率计算

简单收益率
\[R_t = \frac{P_t - P_{t-1}}{P_{t-1}} = \frac{P_t}{P_{t-1}} - 1\]
对数收益率
\[r_t = \ln\left(\frac{P_t}{P_{t-1}}\right) = \ln(P_t) - \ln(P_{t-1})\]
多期累计收益率
\[R_{t,T} = \prod_{i=t}^{T}(1 + R_i) - 1\]
2

波动率计算

历史波动率
\[\sigma = \sqrt{\frac{\sum_{i=1}^{n}(r_i - \bar{r})^2}{n-1}} \times \sqrt{252}\]
指数加权波动率（EWMA）
\[\sigma_t^2 = \lambda \sigma_{t-1}^2 + (1-\lambda)r_t^2\]
3

技术指标

RSI
\[RSI = 100 - \frac{100}{1 + RS}\]

其中 \(RS = \frac{平均上涨幅度}{平均下跌幅度}\)

MACD
\[MACD = EMA_{12} - EMA_{26}\]
\[Signal = EMA_9(MACD)\]
布林带
\[Upper = MA_{20} + 2\sigma\]
\[Lower = MA_{20} - 2\sigma\]
4

财务比率

杜邦分解
\[ROE = \frac{净利润}{销售收入} \times \frac{销售收入}{总资产} \times \frac{总资产}{股东权益}\]
\[ROE = 净利率 \times 资产周转率 \times 权益乘数\]

class FactorEngine:
    """因子计算引擎"""

    def __init__(self):
        self.factor_configs = self._load_factor_configs()

    def calculate_all_factors(self, price_df, financial_df, date):
        """计算所有因子"""
        results = []

        # 价量因子
        pv_factors = self._calculate_price_volume_factors(price_df)
        results.append(pv_factors)

        # 财务因子
        fin_factors = self._calculate_financial_factors(financial_df)
        results.append(fin_factors)

        # 合并并标准化
        all_factors = pd.concat(results, axis=1)
        standardized = self._standardize_factors(all_factors)

        # 转换为长格式
        factor_long = standardized.reset_index().melt(
            id_vars=['stock_code'],
            var_name='factor_name',
            value_name='factor_value'
        )
        factor_long['trade_date'] = date

        return factor_long

    def _calculate_price_volume_factors(self, df):
        """计算价量因子"""
        factors = pd.DataFrame(index=df['stock_code'].unique())

        for code in factors.index:
            stock_data = df[df['stock_code'] == code].copy()

            if len(stock_data) < 20:
                continue

            # 动量因子
            factors.loc[code, 'momentum_20'] = \
                stock_data['close'].iloc[-1] / stock_data['close'].iloc[-20] - 1

            # 波动率因子
            returns = stock_data['close'].pct_change()
            factors.loc[code, 'volatility_20'] = returns.tail(20).std() * np.sqrt(252)

            # 换手率因子
            if 'turnover_rate' in stock_data.columns:
                factors.loc[code, 'turnover_avg_20'] = \
                    stock_data['turnover_rate'].tail(20).mean()

        return factors

    def _calculate_financial_factors(self, df):
        """计算财务因子"""
        factors = pd.DataFrame(index=df['stock_code'].unique())

        for code in factors.index:
            fin_data = df[df['stock_code'] == code].iloc[-1]

            # ROE
            if fin_data['equity'] > 0:
                factors.loc[code, 'roe'] = fin_data['net_income'] / fin_data['equity']

            # 资产负债率
            if fin_data['total_assets'] > 0:
                factors.loc[code, 'debt_ratio'] = \
                    fin_data['total_liabilities'] / fin_data['total_assets']

            # 营收增长率
            if 'revenue_yoy' in fin_data:
                factors.loc[code, 'revenue_growth'] = fin_data['revenue_yoy']

        return factors

    def _standardize_factors(self, df):
        """因子标准化"""
        for col in df.columns:
            # 缩尾处理
            lower = df[col].quantile(0.01)
            upper = df[col].quantile(0.99)
            df[col] = df[col].clip(lower, upper)

            # Z-score标准化
            df[col] = (df[col] - df[col].mean()) / df[col].std()

        return df

def run_scheduler():
    db = FactorDatabase()

    # 每天18:00执行日度更新
    schedule.every().day.at("18:00").do(db.daily_update)

    while True:
        schedule.run_pending()
        time.sleep(60)

import redis
from kafka import KafkaConsumer, KafkaProducer
import json

class RealtimeDataPipeline:
    """实时数据处理管道"""

    def __init__(self):
        self.redis_client = redis.Redis(host='localhost', port=6379)
        self.kafka_consumer = KafkaConsumer(
            'tick_data',
            bootstrap_servers=['localhost:9092'],
            value_deserializer=lambda x: json.loads(x.decode('utf-8'))
        )
        self.kafka_producer = KafkaProducer(
            bootstrap_servers=['localhost:9092'],
            value_serializer=lambda x: json.dumps(x).encode('utf-8')
        )

    def process_tick(self, tick):
        """处理单条tick数据"""
        stock_code = tick['stock_code']
        price = tick['price']
        volume = tick['volume']
        timestamp = tick['timestamp']

        # 更新Redis缓存
        self._update_cache(stock_code, tick)

        # 计算实时指标
        indicators = self._calculate_realtime_indicators(stock_code)

        # 发布计算结果
        self.kafka_producer.send('indicators', indicators)

        return indicators

    def _update_cache(self, stock_code, tick):
        """更新缓存数据"""
        # 保存最新价格
        self.redis_client.hset(f'latest:{stock_code}', mapping={
            'price': tick['price'],
            'volume': tick['volume'],
            'timestamp': tick['timestamp']
        })

        # 保存分钟K线（保留最近1000条）
        minute_key = f"minute:{stock_code}"
        self.redis_client.lpush(minute_key, json.dumps(tick))
        self.redis_client.ltrim(minute_key, 0, 999)

    def _calculate_realtime_indicators(self, stock_code):
        """计算实时指标"""
        # 获取历史数据
        minute_data = self.redis_client.lrange(f"minute:{stock_code}", 0, 59)
        prices = [json.loads(d)['price'] for d in minute_data]

        if len(prices) < 20:
            return {}

        prices = np.array(prices[::-1])  # 时间顺序

        indicators = {
            'stock_code': stock_code,
            'ma5': np.mean(prices[-5:]),
            'ma20': np.mean(prices[-20:]),
            'momentum': (prices[-1] / prices[-20] - 1) * 100,
            'volatility': np.std(np.diff(prices) / prices[:-1]) * np.sqrt(252 * 240)
        }

        return indicators

    def run(self):
        """运行数据处理"""
        print("开始处理实时数据...")
        for message in self.kafka_consumer:
            tick = message.value
            indicators = self.process_tick(tick)
            print(f"处理完成: {tick['stock_code']} @ {tick['price']}")

class SatelliteDataProcessor:
    """卫星图像数据处理"""

    def __init__(self):
        # 加载预训练的目标检测模型
        self.model = models.detection.fasterrcnn_resnet50_fpn(pretrained=True)
        self.model.eval()

        self.transform = transforms.Compose([
            transforms.ToTensor()
        ])

    def count_cars_in_parking_lot(self, image_path):
        """统计停车场车辆数量"""
        # 加载图像
        image = Image.open(image_path).convert('RGB')
        image_tensor = self.transform(image)

        # 目标检测
        with torch.no_grad():
            predictions = self.model([image_tensor])

        # 筛选车辆（类别ID=3为car）
        boxes = predictions[0]['boxes']
        labels = predictions[0]['labels']
        scores = predictions[0]['scores']

        car_count = ((labels == 3) & (scores > 0.5)).sum().item()

        return car_count

    def calculate_parking_lot_occupancy(self, image_path, total_spots):
        """计算停车场占用率"""
        car_count = self.count_cars_in_parking_lot(image_path)
        occupancy = car_count / total_spots
        return occupancy, car_count

    def analyze_time_series(self, image_paths, dates, total_spots):
        """分析时间序列图像"""
        results = []

        for image_path, date in zip(image_paths, dates):
            occupancy, car_count = self.calculate_parking_lot_occupancy(
                image_path, total_spots
            )
            results.append({
                'date': date,
                'car_count': car_count,
                'occupancy': occupancy
            })

        return pd.DataFrame(results)

Q1: 如何处理股票数据的前复权和后复权？
答案
python
def adjust_price(df, adj_factor, method='forward'):
    """
    复权处理

    forward: 前复权，以最新数据为基准向前调整
    backward: 后复权，以最早数据为基准向后调整
    """
    if method == 'forward':
        # 前复权：当前价格不变，历史价格调整
        latest_factor = adj_factor.iloc[-1]
        adj_ratio = adj_factor / latest_factor
    else:
        # 后复权：历史价格不变，当前价格调整
        first_factor = adj_factor.iloc[0]
        adj_ratio = adj_factor / first_factor

    for col in ['open', 'high', 'low', 'close']:
        df[col] = df[col] * adj_ratio

    # 成交量反向调整
    df['volume'] = df['volume'] / adj_ratio

    return df
选择建议
•前复权：适合分析当前股价走势，与当前价格对比
•后复权：适合计算真实收益率，回测策略

def merge_data_sources(price_df, financial_df, news_df, date_col='date'):
    """
    合并多数据源
    """
    # 确保日期格式一致
    for df in [price_df, financial_df, news_df]:
        df[date_col] = pd.to_datetime(df[date_col])

    # 价格数据：日度
    # 财务数据：季度 -> 向前填充到日度
    financial_daily = financial_df.set_index(date_col).resample('D').ffill()

    # 新闻数据：不定期 -> 按日聚合
    news_daily = news_df.groupby(date_col).agg({
        'sentiment': 'mean',
        'news_count': 'sum'
    })

    # 合并
    merged = price_df.merge(
        financial_daily, on=[date_col, 'stock_code'], how='left'
    ).merge(
        news_daily, on=date_col, how='left'
    )

    return merged

class PoliteSpider:
    def __init__(self, delay_range=(1, 3)):
        self.delay_range = delay_range

    def fetch(self, url):
        # 随机延迟
        time.sleep(random.uniform(*self.delay_range))

        # 检查robots.txt
        if not self.is_allowed(url):
            return None

        response = requests.get(url, headers=self.headers)
        return response

    def is_allowed(self, url):
        """检查是否允许爬取"""
        from urllib.robotparser import RobotFileParser
        rp = RobotFileParser()
        rp.set_url(f"{urlparse(url).scheme}://{urlparse(url).netloc}/robots.txt")
        rp.read()
        return rp.can_fetch("*", url)

class DataQualityChecker:
    """数据质量检查"""

    def __init__(self, df):
        self.df = df
        self.issues = []

    def check_completeness(self, required_columns):
        """检查完整性"""
        for col in required_columns:
            if col not in self.df.columns:
                self.issues.append(f"缺少列: {col}")
            else:
                missing_rate = self.df[col].isna().mean()
                if missing_rate > 0.1:
                    self.issues.append(f"列 {col} 缺失率过高: {missing_rate:.1%}")
        return self

    def check_consistency(self):
        """检查一致性"""
        # 检查价格关系
        if all(col in self.df.columns for col in ['high', 'low', 'open', 'close']):
            invalid = (self.df['high'] < self.df['low']).sum()
            if invalid > 0:
                self.issues.append(f"高低价异常: {invalid}条")
        return self

    def check_timeliness(self, date_col, max_delay_days=1):
        """检查时效性"""
        if date_col in self.df.columns:
            latest_date = pd.to_datetime(self.df[date_col]).max()
            delay = (pd.Timestamp.now() - latest_date).days
            if delay > max_delay_days:
                self.issues.append(f"数据延迟: {delay}天")
        return self

    def report(self):
        """生成质量报告"""
        return {
            'total_rows': len(self.df),
            'issues': self.issues,
            'passed': len(self.issues) == 0
        }

class RealtimeDataBuffer:
    """实时数据缓冲处理"""

    def __init__(self, max_delay_ms=1000):
        self.max_delay_ms = max_delay_ms
        self.buffer = {}
        self.timestamps = {}

    def add_data(self, stock_code, data, timestamp):
        """添加数据并检查延迟"""
        current_time = time.time() * 1000
        delay = current_time - timestamp

        if delay > self.max_delay_ms:
            # 数据延迟过大，标记警告
            data['delayed'] = True
            data['delay_ms'] = delay

        self.buffer[stock_code] = data
        self.timestamps[stock_code] = timestamp

    def get_latest(self, stock_code, max_age_ms=5000):
        """获取最新数据"""
        if stock_code not in self.buffer:
            return None

        age = time.time() * 1000 - self.timestamps[stock_code]
        if age > max_age_ms:
            return None  # 数据过期

        return self.buffer[stock_code]

def align_multi_market_data(data_dict, reference_calendar):
    """
    对齐多个市场的数据

    data_dict: {'A股': df1, '港股': df2, '美股': df3}
    reference_calendar: 参考交易日历
    """
    aligned_data = {}

    for market, df in data_dict.items():
        # 转换时区
        if market == '美股':
            df['datetime'] = df['datetime'].dt.tz_convert('Asia/Shanghai')

        # 重采样到日度
        daily = df.set_index('datetime').resample('D').last()

        # 对齐到参考日历
        aligned = daily.reindex(reference_calendar, method='ffill')

        aligned_data[market] = aligned

    return aligned_data
