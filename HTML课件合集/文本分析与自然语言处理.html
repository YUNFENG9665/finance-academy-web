import jieba
import jieba.posseg as pseg
import re
from collections import Counter

class ChineseTextProcessor:
    """中文金融文本处理器"""

    def __init__(self):
        # 加载金融词典
        self._load_financial_dict()
        # 加载停用词
        self.stopwords = self._load_stopwords()

    def _load_financial_dict(self):
        """加载金融专业词典"""
        financial_terms = [
            '市盈率', '市净率', '资产负债率', '流动比率',
            '毛利率', '净利率', 'ROE', 'ROA', 'EPS',
            '利空', '利好', '涨停', '跌停', '龙虎榜',
            '北向资金', '融资融券', '大宗交易', '减持',
            '增持', '回购', '配股', '定增', '可转债'
        ]
        for term in financial_terms:
            jieba.add_word(term)

    def _load_stopwords(self):
        """加载停用词"""
        stopwords = set(['的', '了', '是', '在', '和', '与',
                        '或', '等', '及', '对于', '关于'])
        return stopwords

    def preprocess(self, text):
        """完整预处理流程"""
        # 1. 清理文本
        text = self.clean_text(text)
        # 2. 分词
        words = self.tokenize(text)
        # 3. 去停用词
        words = self.remove_stopwords(words)
        return words

    def clean_text(self, text):
        """清理文本"""
        # 去除HTML标签
        text = re.sub(r'<[^>]+>', '', text)
        # 去除多余空白
        text = re.sub(r'\s+', ' ', text)
        # 保留中文、英文、数字
        text = re.sub(r'[^\u4e00-\u9fa5a-zA-Z0-9\.\%\-]', ' ', text)
        return text.strip()

    def tokenize(self, text, with_pos=False):
        """分词"""
        if with_pos:
            return list(pseg.cut(text))
        return list(jieba.cut(text))

    def remove_stopwords(self, words):
        """去除停用词"""
        return [w for w in words if w not in self.stopwords and len(w) > 1]

    def extract_keywords(self, text, topk=10):
        """提取关键词"""
        import jieba.analyse
        return jieba.analyse.extract_tags(text, topK=topk, withWeight=True)

    def extract_entities(self, text):
        """提取命名实体（简单规则）"""
        entities = {
            'companies': [],
            'persons': [],
            'amounts': [],
            'dates': []
        }

        # 金额提取
        amount_pattern = r'[\d,]+\.?\d*[万亿]?元'
        entities['amounts'] = re.findall(amount_pattern, text)

        # 日期提取
        date_pattern = r'\d{4}年\d{1,2}月\d{1,2}日|\d{4}-\d{2}-\d{2}'
        entities['dates'] = re.findall(date_pattern, text)

        return entities

import nltk
from nltk.tokenize import word_tokenize, sent_tokenize
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer
import spacy

class EnglishTextProcessor:
    """英文金融文本处理器"""

    def __init__(self):
        nltk.download('punkt', quiet=True)
        nltk.download('stopwords', quiet=True)
        nltk.download('wordnet', quiet=True)

        self.stop_words = set(stopwords.words('english'))
        self.lemmatizer = WordNetLemmatizer()
        self.nlp = spacy.load('en_core_web_sm')

    def preprocess(self, text):
        """预处理"""
        # 小写化
        text = text.lower()
        # 分词
        tokens = word_tokenize(text)
        # 去停用词和词形还原
        tokens = [self.lemmatizer.lemmatize(t)
                  for t in tokens
                  if t.isalnum() and t not in self.stop_words]
        return tokens

    def extract_entities_spacy(self, text):
        """使用SpaCy提取命名实体"""
        doc = self.nlp(text)
        entities = {
            'ORG': [],      # 组织/公司
            'PERSON': [],   # 人名
            'MONEY': [],    # 金额
            'DATE': [],     # 日期
            'PERCENT': []   # 百分比
        }
        for ent in doc.ents:
            if ent.label_ in entities:
                entities[ent.label_].append(ent.text)
        return entities

class DictionarySentiment:
    """基于词典的情感分析"""

    def __init__(self):
        # Loughran-McDonald金融情感词典
        self.positive_words = self._load_lm_positive()
        self.negative_words = self._load_lm_negative()
        self.uncertain_words = self._load_lm_uncertain()
        self.litigious_words = self._load_lm_litigious()

        # 中文金融情感词典
        self.cn_positive = set(['利好', '增长', '盈利', '突破', '上涨',
                                '创新高', '超预期', '业绩靓丽', '强劲'])
        self.cn_negative = set(['利空', '下跌', '亏损', '下滑', '风险',
                                '暴跌', '违约', '暴雷', '减持', '清仓'])

    def _load_lm_positive(self):
        """加载LM正面词典"""
        # 实际应从文件加载
        return set(['achieve', 'benefit', 'excellent', 'gain',
                   'improve', 'positive', 'profit', 'success'])

    def _load_lm_negative(self):
        """加载LM负面词典"""
        return set(['adverse', 'decline', 'deficit', 'loss',
                   'negative', 'risk', 'weak', 'worse'])

    def analyze(self, text, language='en'):
        """分析文本情感"""
        if language == 'en':
            return self._analyze_english(text)
        else:
            return self._analyze_chinese(text)

    def _analyze_english(self, text):
        """英文情感分析"""
        words = text.lower().split()
        total = len(words)

        pos_count = sum(1 for w in words if w in self.positive_words)
        neg_count = sum(1 for w in words if w in self.negative_words)
        unc_count = sum(1 for w in words if w in self.uncertain_words)

        # 计算情感得分
        sentiment_score = (pos_count - neg_count) / total if total > 0 else 0

        return {
            'sentiment_score': sentiment_score,
            'positive_ratio': pos_count / total if total > 0 else 0,
            'negative_ratio': neg_count / total if total > 0 else 0,
            'uncertainty_ratio': unc_count / total if total > 0 else 0,
            'classification': self._classify(sentiment_score)
        }

    def _analyze_chinese(self, text):
        """中文情感分析"""
        import jieba
        words = list(jieba.cut(text))
        total = len(words)

        pos_count = sum(1 for w in words if w in self.cn_positive)
        neg_count = sum(1 for w in words if w in self.cn_negative)

        sentiment_score = (pos_count - neg_count) / total if total > 0 else 0

        return {
            'sentiment_score': sentiment_score,
            'positive_count': pos_count,
            'negative_count': neg_count,
            'classification': self._classify(sentiment_score)
        }

    def _classify(self, score):
        if score > 0.02:
            return 'positive'
        elif score < -0.02:
            return 'negative'
        else:
            return 'neutral'

深度学习情感分析

python
import torch
from transformers import AutoTokenizer, AutoModelForSequenceClassification

class DeepSentimentAnalyzer:
    """基于深度学习的情感分析"""

    def __init__(self, model_name='yiyanghkust/finbert-tone'):
        """
        加载预训练模型

        推荐模型:
        - yiyanghkust/finbert-tone: 金融情感
        - ProsusAI/finbert: 金融情感
        - nlptown/bert-base-multilingual-uncased-sentiment: 多语言情感
        - uer/roberta-base-finetuned-jd-binary-chinese: 中文情感
        """
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(model_name)
        self.model.eval()

        # 标签映射
        self.label_map = {0: 'negative', 1: 'neutral', 2: 'positive'}

    def analyze(self, text):
        """分析单条文本"""
        inputs = self.tokenizer(
            text,
            return_tensors='pt',
            truncation=True,
            max_length=512,
            padding=True
        )

        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.softmax(outputs.logits, dim=-1)

        pred_label = torch.argmax(probs, dim=-1).item()
        confidence = probs[0][pred_label].item()

        return {
            'sentiment': self.label_map.get(pred_label, 'neutral'),
            'confidence': confidence,
            'probabilities': {
                'negative': probs[0][0].item(),
                'neutral': probs[0][1].item(),
                'positive': probs[0][2].item()
            }
        }

    def analyze_batch(self, texts, batch_size=32):
        """批量分析"""
        results = []
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            inputs = self.tokenizer(
                batch,
                return_tensors='pt',
                truncation=True,
                max_length=512,
                padding=True
            )

            with torch.no_grad():
                outputs = self.model(**inputs)
                probs = torch.softmax(outputs.logits, dim=-1)

            for j, prob in enumerate(probs):
                pred_label = torch.argmax(prob).item()
                results.append({
                    'text': batch[j][:50] + '...',
                    'sentiment': self.label_map.get(pred_label, 'neutral'),
                    'confidence': prob[pred_label].item()
                })

        return results

import re

class RuleBasedNER:
    """基于规则的命名实体识别"""

    def __init__(self):
        self.patterns = {
            'stock_code': r'[036]\d{5}\.[A-Z]{2}|[A-Z]{1,5}',
            'money_cn': r'[\d,]+\.?\d*[万亿]?元',
            'money_en': r'\$[\d,]+\.?\d*\s*(million|billion|M|B)?',
            'percent': r'-?\d+\.?\d*\%',
            'date_cn': r'\d{4}年\d{1,2}月\d{1,2}日',
            'date_en': r'\d{4}-\d{2}-\d{2}|\w+ \d{1,2}, \d{4}',
            'ratio': r'\d+\.?\d*[xX倍]'
        }

    def extract(self, text):
        """提取所有实体"""
        entities = {}
        for entity_type, pattern in self.patterns.items():
            matches = re.findall(pattern, text)
            if matches:
                entities[entity_type] = matches
        return entities

    def extract_company_names(self, text):
        """提取公司名（中文）"""
        # 常见公司名后缀
        suffixes = ['股份', '集团', '公司', '有限', '控股', '科技', '银行', '证券']
        companies = []

        for suffix in suffixes:
            # 匹配包含后缀的词组
            pattern = rf'[\u4e00-\u9fa5]{{2,10}}{suffix}'
            matches = re.findall(pattern, text)
            companies.extend(matches)

        return list(set(companies))

from transformers import AutoTokenizer, AutoModelForTokenClassification
from transformers import pipeline

class DeepNER:
    """基于深度学习的NER"""

    def __init__(self, model_name='uer/roberta-base-finetuned-cluener2020-chinese'):
        """
        加载NER模型

        推荐模型:
        - uer/roberta-base-finetuned-cluener2020-chinese: 中文NER
        - dslim/bert-base-NER: 英文NER
        - xlm-roberta-large-finetuned-conll03-english: 多语言NER
        """
        self.ner_pipeline = pipeline(
            'ner',
            model=model_name,
            tokenizer=model_name,
            aggregation_strategy='simple'
        )

    def extract(self, text):
        """提取命名实体"""
        results = self.ner_pipeline(text)

        # 整理结果
        entities = {}
        for ent in results:
            entity_type = ent['entity_group']
            if entity_type not in entities:
                entities[entity_type] = []
            entities[entity_type].append({
                'text': ent['word'],
                'score': ent['score'],
                'start': ent['start'],
                'end': ent['end']
            })

        return entities

    def extract_financial_entities(self, text):
        """提取金融相关实体"""
        entities = self.extract(text)

        # 后处理：标准化金额
        if 'MONEY' in entities:
            for ent in entities['MONEY']:
                ent['normalized'] = self._normalize_money(ent['text'])

        return entities

    def _normalize_money(self, money_str):
        """标准化金额"""
        # 移除逗号
        clean = money_str.replace(',', '').replace(' ', '')

        # 识别单位
        multiplier = 1
        if '万' in clean or 'M' in clean.upper():
            multiplier = 10000
        elif '亿' in clean or 'B' in clean.upper():
            multiplier = 100000000

        # 提取数字
        numbers = re.findall(r'[\d.]+', clean)
        if numbers:
            return float(numbers[0]) * multiplier
        return None

import re

class TemplateRelationExtractor:
    """基于模板的关系抽取"""

    def __init__(self):
        self.templates = {
            'investment': [
                r'([\u4e00-\u9fa5]+)投资([\u4e00-\u9fa5]+)',
                r'([\u4e00-\u9fa5]+)入股([\u4e00-\u9fa5]+)',
                r'([\u4e00-\u9fa5]+)参股([\u4e00-\u9fa5]+)'
            ],
            'acquisition': [
                r'([\u4e00-\u9fa5]+)收购([\u4e00-\u9fa5]+)',
                r'([\u4e00-\u9fa5]+)并购([\u4e00-\u9fa5]+)',
                r'([\u4e00-\u9fa5]+)要约收购([\u4e00-\u9fa5]+)'
            ],
            'shareholding': [
                r'([\u4e00-\u9fa5]+)持有([\u4e00-\u9fa5]+)(\d+\.?\d*)[%％]',
                r'([\u4e00-\u9fa5]+)持股([\u4e00-\u9fa5]+)(\d+\.?\d*)[%％]'
            ],
            'supply': [
                r'([\u4e00-\u9fa5]+)是([\u4e00-\u9fa5]+)的供应商',
                r'([\u4e00-\u9fa5]+)为([\u4e00-\u9fa5]+)供货'
            ],
            'executive': [
                r'([\u4e00-\u9fa5]{2,4})担任([\u4e00-\u9fa5]+)(董事长|CEO|总裁|总经理)',
                r'([\u4e00-\u9fa5]{2,4})任([\u4e00-\u9fa5]+)(董事长|CEO|总裁|总经理)'
            ]
        }

    def extract(self, text):
        """提取所有关系"""
        relations = []

        for rel_type, patterns in self.templates.items():
            for pattern in patterns:
                matches = re.finditer(pattern, text)
                for match in matches:
                    groups = match.groups()
                    relation = {
                        'type': rel_type,
                        'subject': groups[0],
                        'object': groups[1],
                        'text': match.group()
                    }
                    if len(groups) > 2:
                        relation['value'] = groups[2]
                    relations.append(relation)

        return relations

from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch

class DeepRelationExtractor:
    """基于深度学习的关系抽取"""

    def __init__(self):
        # 使用预训练的关系分类模型
        self.tokenizer = AutoTokenizer.from_pretrained('bert-base-chinese')
        # 这里需要微调过的模型，示例使用基础模型
        self.model = AutoModelForSequenceClassification.from_pretrained(
            'bert-base-chinese',
            num_labels=10  # 关系类型数
        )
        self.relation_labels = [
            'investment', 'acquisition', 'shareholding', 'supply',
            'competition', 'executive', 'cooperation', 'subsidiary',
            'customer', 'other'
        ]

    def extract(self, text, entity1, entity2):
        """
        给定两个实体，判断关系类型

        Args:
            text: 包含两个实体的文本
            entity1: 主体实体
            entity2: 客体实体
        """
        # 构造输入：[CLS] entity1 [SEP] entity2 [SEP] text [SEP]
        input_text = f"{entity1} [SEP] {entity2} [SEP] {text}"

        inputs = self.tokenizer(
            input_text,
            return_tensors='pt',
            truncation=True,
            max_length=256
        )

        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.softmax(outputs.logits, dim=-1)

        pred_idx = torch.argmax(probs).item()

        return {
            'subject': entity1,
            'object': entity2,
            'relation': self.relation_labels[pred_idx],
            'confidence': probs[0][pred_idx].item()
        }

            max_features=5000,
            ngram_range=(1, 2)
        )

        if model_type == 'nb':
            self.classifier = MultinomialNB()
        elif model_type == 'svm':
            self.classifier = SVC(kernel='linear', probability=True)
        elif model_type == 'rf':
            self.classifier = RandomForestClassifier(n_estimators=100)

        self.pipeline = Pipeline([
            ('tfidf', self.vectorizer),
            ('clf', self.classifier)
        ])

    def _chinese_tokenizer(self, text):
        """中文分词器"""
        return list(jieba.cut(text))

    def train(self, texts, labels):
        """训练模型"""
        self.pipeline.fit(texts, labels)

    def predict(self, text):
        """预测"""
        pred = self.pipeline.predict([text])[0]
        proba = self.pipeline.predict_proba([text])[0]
        return {
            'prediction': pred,
            'probabilities': dict(zip(self.pipeline.classes_, proba))
        }

    def evaluate(self, texts, labels, cv=5):
        """交叉验证评估"""
        scores = cross_val_score(self.pipeline, texts, labels, cv=cv)
        return {
            'mean_accuracy': scores.mean(),
            'std': scores.std()
        }

from datasets import Dataset
import torch

class DeepTextClassifier:
    """基于深度学习的文本分类"""

    def __init__(self, model_name='bert-base-chinese', num_labels=5):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            model_name,
            num_labels=num_labels
        )
        self.label_map = {}

    def prepare_data(self, texts, labels):
        """准备训练数据"""
        # 构建标签映射
        unique_labels = list(set(labels))
        self.label_map = {label: i for i, label in enumerate(unique_labels)}
        self.reverse_label_map = {i: label for label, i in self.label_map.items()}

        # 转换标签
        numeric_labels = [self.label_map[l] for l in labels]

        # 创建数据集
        dataset = Dataset.from_dict({
            'text': texts,
            'label': numeric_labels
        })

        # tokenize
        def tokenize_function(examples):
            return self.tokenizer(
                examples['text'],
                truncation=True,
                padding='max_length',
                max_length=256
            )

        tokenized_dataset = dataset.map(tokenize_function, batched=True)
        return tokenized_dataset

    def train(self, train_texts, train_labels, val_texts=None, val_labels=None):
        """训练模型"""
        train_dataset = self.prepare_data(train_texts, train_labels)

        val_dataset = None
        if val_texts and val_labels:
            val_dataset = self.prepare_data(val_texts, val_labels)

        training_args = TrainingArguments(
            output_dir='./results',
            num_train_epochs=3,
            per_device_train_batch_size=16,
            per_device_eval_batch_size=16,
            warmup_steps=500,
            weight_decay=0.01,
            logging_dir='./logs',
            evaluation_strategy='epoch' if val_dataset else 'no'
        )

        trainer = Trainer(
            model=self.model,
            args=training_args,
            train_dataset=train_dataset,
            eval_dataset=val_dataset
        )

        trainer.train()

    def predict(self, text):
        """预测"""
        inputs = self.tokenizer(
            text,
            return_tensors='pt',
            truncation=True,
            max_length=256
        )

        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.softmax(outputs.logits, dim=-1)

        pred_idx = torch.argmax(probs).item()

        return {
            'prediction': self.reverse_label_map[pred_idx],
            'confidence': probs[0][pred_idx].item(),
            'all_probabilities': {
                self.reverse_label_map[i]: prob.item()
                for i, prob in enumerate(probs[0])
            }
        }

from sentence_transformers import SentenceTransformer
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

class TextEmbedding:
    """文本向量化"""

    def __init__(self, model_name='paraphrase-multilingual-MiniLM-L12-v2'):
        """
        加载嵌入模型

        推荐模型:
        - paraphrase-multilingual-MiniLM-L12-v2: 多语言通用
        - shibing624/text2vec-base-chinese: 中文优化
        - all-MiniLM-L6-v2: 英文快速
        - BAAI/bge-large-zh-v1.5: 中文高质量
        """
        self.model = SentenceTransformer(model_name)

    def encode(self, texts, batch_size=32):
        """编码文本"""
        if isinstance(texts, str):
            texts = [texts]
        embeddings = self.model.encode(texts, batch_size=batch_size)
        return embeddings

    def similarity(self, text1, text2):
        """计算文本相似度"""
        emb1 = self.encode(text1)
        emb2 = self.encode(text2)
        return cosine_similarity([emb1], [emb2])[0][0]

    def semantic_search(self, query, documents, top_k=5):
        """语义搜索"""
        query_emb = self.encode(query)
        doc_embs = self.encode(documents)

        # 计算相似度
        similarities = cosine_similarity([query_emb], doc_embs)[0]

        # 排序
        top_indices = np.argsort(similarities)[::-1][:top_k]

        results = []
        for idx in top_indices:
            results.append({
                'document': documents[idx],
                'score': similarities[idx],
                'index': idx
            })

        return results

import faiss
import numpy as np
import pickle

class FinancialDocumentRetrieval:
    """金融文档检索系统"""

    def __init__(self, embedding_model=None):
        self.embedder = embedding_model or TextEmbedding()
        self.index = None
        self.documents = []
        self.metadata = []

    def build_index(self, documents, metadata=None):
        """构建索引"""
        self.documents = documents
        self.metadata = metadata or [{}] * len(documents)

        # 编码文档
        embeddings = self.embedder.encode(documents)
        embeddings = np.array(embeddings).astype('float32')

        # 构建FAISS索引
        dimension = embeddings.shape[1]
        self.index = faiss.IndexFlatIP(dimension)  # 内积相似度
        faiss.normalize_L2(embeddings)  # L2归一化
        self.index.add(embeddings)

    def search(self, query, top_k=10):
        """搜索"""
        query_emb = self.embedder.encode(query)
        query_emb = np.array([query_emb]).astype('float32')
        faiss.normalize_L2(query_emb)

        scores, indices = self.index.search(query_emb, top_k)

        results = []
        for score, idx in zip(scores[0], indices[0]):
            results.append({
                'document': self.documents[idx],
                'score': float(score),
                'metadata': self.metadata[idx]
            })

        return results

    def save(self, path):
        """保存索引"""
        faiss.write_index(self.index, f"{path}/index.faiss")
        with open(f"{path}/data.pkl", 'wb') as f:
            pickle.dump({
                'documents': self.documents,
                'metadata': self.metadata
            }, f)

    def load(self, path):
        """加载索引"""
        self.index = faiss.read_index(f"{path}/index.faiss")
        with open(f"{path}/data.pkl", 'rb') as f:
            data = pickle.load(f)
            self.documents = data['documents']
            self.metadata = data['metadata']

核心公式

1

TF-IDF

词频（TF）
\[TF(t, d) = \frac{词t在文档d中出现的次数}{文档d的总词数}\]
逆文档频率（IDF）
\[IDF(t) = \log\frac{文档总数}{包含词t的文档数 + 1}\]
TF-IDF
\[TF\text{-}IDF(t, d) = TF(t, d) \times IDF(t)\]
2

情感得分

词典法情感得分
\[Sentiment = \frac{正面词数 - 负面词数}{总词数}\]
加权情感得分
\[Sentiment_{weighted} = \frac{\sum_i (s_i \times w_i)}{\sum_i w_i}\]

其中 \(s_i\) 为单条情感得分，\(w_i\) 为权重（如转发数、点赞数）

3

文本相似度

余弦相似度
\[\cos(\theta) = \frac{\vec{A} \cdot \vec{B}}{||\vec{A}|| \times ||\vec{B}||} = \frac{\sum_i A_i B_i}{\sqrt{\sum_i A_i^2} \times \sqrt{\sum_i B_i^2}}\]
4

注意力机制

Self-Attention
\[Attention(Q, K, V) = softmax\left(\frac{QK^T}{\sqrt{d_k}}\right)V\]

import pandas as pd
import numpy as np
from scipy import stats

class NewsSentimentResearch:
    """新闻情绪研究"""

    def __init__(self, sentiment_analyzer):
        self.analyzer = sentiment_analyzer

    def calculate_daily_sentiment(self, news_df, date_col='date', text_col='content'):
        """计算每日每股情绪得分"""
        results = []

        for _, row in news_df.iterrows():
            sentiment = self.analyzer.analyze(row[text_col])
            results.append({
                'date': row[date_col],
                'stock_code': row['stock_code'],
                'sentiment_score': sentiment['sentiment_score'],
                'news_count': 1
            })

        df = pd.DataFrame(results)

        # 按日期和股票聚合
        daily_sentiment = df.groupby(['date', 'stock_code']).agg({
            'sentiment_score': 'mean',
            'news_count': 'sum'
        }).reset_index()

        return daily_sentiment

    def merge_with_returns(self, sentiment_df, price_df):
        """合并情绪和收益数据"""
        # 计算次日收益
        price_df = price_df.sort_values(['stock_code', 'date'])
        price_df['next_return'] = price_df.groupby('stock_code')['close'].pct_change().shift(-1)

        # 合并
        merged = sentiment_df.merge(
            price_df[['date', 'stock_code', 'next_return']],
            on=['date', 'stock_code'],
            how='inner'
        )

        return merged

    def analyze_sentiment_return_relation(self, merged_df):
        """分析情绪与收益关系"""
        # 1. 相关性分析
        correlation = merged_df['sentiment_score'].corr(merged_df['next_return'])

        # 2. 分组检验
        # 按情绪分组
        merged_df['sentiment_group'] = pd.qcut(
            merged_df['sentiment_score'],
            q=5,
            labels=['Q1_Neg', 'Q2', 'Q3', 'Q4', 'Q5_Pos']
        )

        group_returns = merged_df.groupby('sentiment_group')['next_return'].mean()

        # 3. T检验：最正面 vs 最负面
        pos_returns = merged_df[merged_df['sentiment_group'] == 'Q5_Pos']['next_return']
        neg_returns = merged_df[merged_df['sentiment_group'] == 'Q1_Neg']['next_return']
        t_stat, p_value = stats.ttest_ind(pos_returns, neg_returns)

        return {
            'correlation': correlation,
            'group_returns': group_returns.to_dict(),
            't_statistic': t_stat,
            'p_value': p_value,
            'spread': group_returns['Q5_Pos'] - group_returns['Q1_Neg']
        }

    def backtest_sentiment_strategy(self, merged_df, top_n=50):
        """回测情绪策略"""
        results = []

        for date in merged_df['date'].unique():
            daily_data = merged_df[merged_df['date'] == date]

            # 选择情绪最正面的股票
            top_stocks = daily_data.nlargest(top_n, 'sentiment_score')

            # 等权组合收益
            portfolio_return = top_stocks['next_return'].mean()

            results.append({
                'date': date,
                'portfolio_return': portfolio_return,
                'avg_sentiment': top_stocks['sentiment_score'].mean()
            })

        results_df = pd.DataFrame(results)

        # 计算累计收益
        results_df['cumulative_return'] = (1 + results_df['portfolio_return']).cumprod()

        # 计算统计指标
        annual_return = results_df['portfolio_return'].mean() * 252
        annual_vol = results_df['portfolio_return'].std() * np.sqrt(252)
        sharpe = annual_return / annual_vol if annual_vol > 0 else 0

        return {
            'daily_returns': results_df,
            'annual_return': annual_return,
            'annual_volatility': annual_vol,
            'sharpe_ratio': sharpe
        }

class EarningsCallAnalyzer:
    """财报电话会议分析"""

    def __init__(self):
        self.sentiment_analyzer = FinBERTSentiment()
        self.text_processor = EnglishTextProcessor()

    def analyze_transcript(self, transcript):
        """分析电话会议文本"""
        # 分割问答部分
        sections = self._split_sections(transcript)

        results = {
            'prepared_remarks': self._analyze_section(sections.get('prepared', '')),
            'qa_section': self._analyze_section(sections.get('qa', '')),
            'management_tone': None,
            'analyst_questions': None
        }

        # 分析管理层语气
        if 'prepared' in sections:
            results['management_tone'] = self._analyze_management_tone(sections['prepared'])

        # 分析分析师问题
        if 'qa' in sections:
            results['analyst_questions'] = self._analyze_analyst_questions(sections['qa'])

        return results

    def _split_sections(self, transcript):
        """分割电话会议各部分"""
        sections = {}

        # 简化分割逻辑
        if 'Q&A' in transcript or 'Question' in transcript:
            parts = transcript.split('Q&A')
            if len(parts) >= 2:
                sections['prepared'] = parts[0]
                sections['qa'] = parts[1]
        else:
            sections['prepared'] = transcript

        return sections

    def _analyze_section(self, text):
        """分析文本段落"""
        if not text:
            return None

        sentences = text.split('.')
        sentiments = []

        for sent in sentences:
            if len(sent.strip()) > 20:
                result = self.sentiment_analyzer.predict(sent)
                sentiments.append(result)

        if not sentiments:
            return None

        # 统计
        labels = [s['label'] for s in sentiments]

        return {
            'total_sentences': len(sentiments),
            'positive_ratio': labels.count('positive') / len(labels),
            'negative_ratio': labels.count('negative') / len(labels),
            'avg_confidence': np.mean([s['score'] for s in sentiments])
        }

    def _analyze_management_tone(self, text):
        """分析管理层语气"""
        # 关键词分析
        confidence_words = ['confident', 'strong', 'robust', 'excellent', 'record']
        hedging_words = ['may', 'might', 'could', 'possibly', 'uncertain']

        text_lower = text.lower()

        confidence_count = sum(1 for w in confidence_words if w in text_lower)
        hedging_count = sum(1 for w in hedging_words if w in text_lower)

        return {
            'confidence_score': confidence_count / (confidence_count + hedging_count + 1),
            'confidence_words': confidence_count,
            'hedging_words': hedging_count
        }

    def _analyze_analyst_questions(self, text):
        """分析分析师问题"""
        # 识别问题（以问号结尾）
        questions = [s.strip() for s in text.split('?') if len(s.strip()) > 20]

        # 分类问题主题
        topics = {
            'guidance': 0,
            'margin': 0,
            'growth': 0,
            'competition': 0,
            'other': 0
        }

        for q in questions:
            q_lower = q.lower()
            if any(w in q_lower for w in ['guidance', 'outlook', 'expect']):
                topics['guidance'] += 1
            elif any(w in q_lower for w in ['margin', 'profit', 'cost']):
                topics['margin'] += 1
            elif any(w in q_lower for w in ['growth', 'increase', 'expand']):
                topics['growth'] += 1
            elif any(w in q_lower for w in ['competition', 'competitor', 'market share']):
                topics['competition'] += 1
            else:
                topics['other'] += 1

        return {
            'total_questions': len(questions),
            'topics': topics
        }

    def compare_quarters(self, current_transcript, previous_transcript):
        """对比季度变化"""
        current = self.analyze_transcript(current_transcript)
        previous = self.analyze_transcript(previous_transcript)

        comparison = {}

        if current['prepared_remarks'] and previous['prepared_remarks']:
            comparison['sentiment_change'] = (
                current['prepared_remarks']['positive_ratio'] -
                previous['prepared_remarks']['positive_ratio']
            )

        if current['management_tone'] and previous['management_tone']:
            comparison['confidence_change'] = (
                current['management_tone']['confidence_score'] -
                previous['management_tone']['confidence_score']
            )

        return comparison

class ResearchReportExtractor:

    def _extract_rating(self, text):
        """提取投资评级"""
        rating_patterns = {
            'buy': r'(买入|强烈推荐|推荐|Buy|Strong Buy)',
            'hold': r'(增持|谨慎推荐|中性|Hold|Neutral)',
            'sell': r'(减持|卖出|回避|Sell|Underperform)'
        }

        for rating, pattern in rating_patterns.items():
            if re.search(pattern, text, re.IGNORECASE):
                return rating

        return 'unknown'

    def _extract_target_price(self, text):
        """提取目标价"""
        patterns = [
            r'目标价[：:]\s*(\d+\.?\d*)\s*元',
            r'目标价位?\s*(\d+\.?\d*)\s*元',
            r'Target Price[：:]\s*\$?(\d+\.?\d*)',
            r'(\d+\.?\d*)\s*元.*目标价'
        ]

        for pattern in patterns:
            match = re.search(pattern, text)
            if match:
                return float(match.group(1))

        return None

    def _extract_key_points(self, text):
        """提取核心观点"""
        key_points = []

        # 方法1：寻找列表格式的观点
        list_patterns = [
            r'[（\(]?\d[）\)][、.]\s*(.+?)(?=[（\(]?\d[）\)]|$)',
            r'[•●]\s*(.+?)(?=[•●]|$)'
        ]

        for pattern in list_patterns:
            matches = re.findall(pattern, text)
            key_points.extend(matches[:5])  # 取前5个

        # 方法2：提取含关键词的句子
        if not key_points:
            keywords = ['核心', '关键', '亮点', '优势', '创新']
            sentences = text.split('。')
            for sent in sentences:
                if any(kw in sent for kw in keywords):
                    key_points.append(sent.strip())

        return key_points[:5]

    def _extract_risks(self, text):
        """提取风险提示"""
        risks = []

        # 定位风险提示部分
        risk_section_pattern = r'风险提示[：:](.*?)(?=\n\n|\Z)'
        risk_match = re.search(risk_section_pattern, text, re.DOTALL)

        if risk_match:
            risk_text = risk_match.group(1)
            # 提取风险点
            risk_items = re.findall(r'[（\(]?\d[）\)][、.]\s*(.+?)(?=[（\(]?\d[）\)]|$)', risk_text)
            risks.extend(risk_items)

        if not risks:
            # 备用方法
            risk_keywords = ['风险', '不确定', '挑战', '下滑', '竞争']
            sentences = text.split('。')
            for sent in sentences:
                if any(kw in sent for kw in risk_keywords):
                    risks.append(sent.strip())

        return risks[:5]

    def _extract_forecasts(self, text):
        """提取财务预测"""
        forecasts = {}

        # 收入预测
        revenue_pattern = r'预计.*?营[业收].*?(\d+\.?\d*)[亿万]'
        revenue_match = re.search(revenue_pattern, text)
        if revenue_match:
            forecasts['revenue'] = float(revenue_match.group(1))

        # 利润预测
        profit_pattern = r'预计.*?净利润.*?(\d+\.?\d*)[亿万]'
        profit_match = re.search(profit_pattern, text)
        if profit_match:
            forecasts['net_profit'] = float(profit_match.group(1))

        # EPS预测
        eps_pattern = r'EPS.*?(\d+\.?\d*)\s*元'
        eps_match = re.search(eps_pattern, text)
        if eps_match:
            forecasts['eps'] = float(eps_match.group(1))

        return forecasts

    def _analyze_report_sentiment(self, text):
        """分析研报整体情感"""
        # 抽取关键段落
        key_sections = []

        # 投资建议部分
        invest_pattern = r'投资建议[：:](.*?)(?=\n\n|\Z)'
        invest_match = re.search(invest_pattern, text, re.DOTALL)
        if invest_match:
            key_sections.append(invest_match.group(1)[:500])

        # 核心观点部分
        core_pattern = r'核心观点[：:](.*?)(?=\n\n|\Z)'
        core_match = re.search(core_pattern, text, re.DOTALL)
        if core_match:
            key_sections.append(core_match.group(1)[:500])

        if key_sections:
            combined_text = ' '.join(key_sections)
            return self.sentiment.analyze(combined_text)

        return self.sentiment.analyze(text[:1000])

Q4: 如何处理金融文本中的否定和反讽？
答案
python
class NegationHandler:
    """否定处理"""

    def __init__(self):
        self.negation_words = ['不', '没有', '并非', '未', '无',
                               'not', "n't", 'no', 'never']
        self.negation_window = 3  # 否定词影响范围

    def handle_negation(self, tokens, sentiments):
        """处理否定词对情感的影响"""
        for i, token in enumerate(tokens):
            if token in self.negation_words:
                # 翻转窗口内的情感
                for j in range(i+1, min(i+self.negation_window+1, len(tokens))):
                    if sentiments[j] == 'positive':
                        sentiments[j] = 'negative'
                    elif sentiments[j] == 'negative':
                        sentiments[j] = 'positive'

        return sentiments

def auto_label_stocks(text, stock_list):
    labels = ['O'] * len(text)
    for stock_name, stock_code in stock_list:
        start = text.find(stock_name)
        if start != -1:
            labels[start] = 'B-ORG'
            for i in range(start+1, start+len(stock_name)):
                labels[i] = 'I-ORG'
    return labels
